# pandas_apps

### Install

    git clone https://github.com/evadelzz1/pandas_apps.git

    cd ./pandas_apps

    echo '.env'  >> .gitignore
    echo '.venv' >> .gitignore

    python -m venv .venv

    source .venv/bin/activate

    pip install -r requirements.txt

    python app.py

    deactivate
